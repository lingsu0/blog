[原题链接](https://leetcode.cn/problems/two-sum/)

```java
class Solution {
    public int[] twoSum(int[] nums, int target) {
        // 创建一个哈希表
        Map<Integer , Integer> numtable = new HashMap<Integer , Integer>();
        int n = nums.length;
        for(int i = 0 ; i < n ; ++i){
            // 判断当前数字是否在哈希表中
            if(numtable.containsKey(target - nums[i])){
                // 如果在哈希表中，则返回结果
                return new int[]{numtable.get(target - nums[i]), i};
            }
            // 否则将当前数字加入哈希表
            numtable.put(nums[i],i);
        }
        // 如果没有找到结果，则返回空数组
        return new int[0];
    }
}
```
